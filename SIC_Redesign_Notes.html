    ("Research", "Student Involvement"),

    ("Education, Outreach, and Public Engagement", "Media Content"),
    ("Education, Outreach, and Public Engagement", "Grant and Funding Applications"),
    ("Education, Outreach, and Public Engagement", "Student Involvement"),
    ("Data Archival and Preservation", "Education, Outreach, and Public Engagement"),
    ("LPL Support", "Class Support"),
    ("LPL Support", "Grant and Funding Applications")
]

unique_strings = {string for edge in edges for string in edge}
functions = list(unique_strings)

importance_values = [1, 4, 4, 3, 2, 4, 3, 4, 4, 5, 4, 5]

G = nx.DiGraph()

# Add nodes with labels and sizes

    G.add_node(function, size=importance)

G.add_edges_from(edges)

# Draw the graph
plt.figure(figsize=(18, 12))
pos = nx.spring_layout(G)  # You can use different layout algorithms

nx.draw_networkx_nodes(G, pos, node_size=node_sizes, node_color='skyblue', alpha=0.8)
nx.draw_networkx_labels(G, pos, font_size=10, font_color="black", font_weight="bold",)
nx.draw_networkx_edges(G, pos, edge_color='gray', alpha=0.5, arrows=True)

plt.title('Directed Graph of Space Imagery Center Functions and Relationships')
plt.axis('off')
plt.show()

# Import necessary libraries
import networkx as nx
import matplotlib.pyplot as plt

# Function to create a directed graph
def create_dependency_graph(dependencies):
    G = nx.DiGraph()

    # Add nodes and edges based on dependencies
    for item, dependent_items in dependencies.items():
        G.add_node(item)
        G.add_edges_from((item, dependent_item) for dependent_item in dependent_items)

    return G
